

#pragma once
#include "sessions/session.hpp"
#include "db/db.hpp"
#include "C2Common/types.hpp"

namespace C2::db {

    /**
     * @brief Repository for all CRUD operations on the `clients` table.
     *
     * Follows the repository pattern — all SQL touching `clients` lives here,
     * keeping database logic out of business/networking code.
     *
     * ### Database access
     * Does not hold or accept a `Database` reference. Instead, the private
     * `db()` helper retrieves the connection directly from `Context::getInstance()`
     * on every call. This removes the need to thread a `Database&` through
     * `ClientsManager` and eliminates the declaration-order dependency in `Context`.
     *
     * ### Thread safety
     * No internal synchronisation is performed. If multiple threads share the
     * same `ClientRepository`, the caller is responsible for serialising access.
     *
     * ### Usage
     * @code
     *   CppC2::db::ClientRepository repo;  // no db param needed
     *
     *   repo.add(client);               // on connect
     *   repo.disconnect(id, timepoint); // on graceful disconnect
     *   repo.remove(id);                // hard delete
     * @endcode
     */
    class SessionRepository {
    public:
        SessionRepository() noexcept = default;

        /**
         * @brief Inserts a new client record when a client connects.
         *
         * Persists `id`, `address` (ip:port), `listen_type`, and `connected_at`.
         * Logs and returns false if the prepare or execute step fails.
         *
         * @param client  The connected client to persist.
         * @return true   on successful insert.
         * @return false  on prepare failure or execute failure (e.g. duplicate id).
         */
        bool add(const C2::Sessions::Session& session) noexcept;


        /**
         * @brief Permanently removes a client record from the table.
         *
         * @param id  The unique client id to delete.
         * @return true   if the row was deleted.
         * @return false  on prepare failure or execute failure.
         */
        bool remove(const C2Common::Session::sessionId& id) noexcept;

        /**
         * @brief Stamps `disconnected_at` for a client without deleting the row.
         *
         * Preserves the row for audit/history purposes. Use `remove()` if the
         * record should be fully purged instead.
         *
         * @param id              The client to update.
         * @param disconnectedAt  Timestamp to write into `disconnected_at`.
         * @return true   if the row was updated.
         * @return false  on prepare failure or execute failure.
         */
        bool disconnect(const C2Common::Session::sessionId& id,
                        const C2Common::TimePoint& disconnectedAt) noexcept;

        /**
         * @brief Checks whether a client record exists in the table.
         *
         * Runs `SELECT COUNT(*) FROM clients WHERE id = ?` and returns true
         * if the count is greater than zero.
         *
         * @param id  The client id to look up.
         * @return true   if at least one matching row exists.
         * @return false  if not found, or on prepare/query failure.
         */
        bool exists(const C2Common::Session::sessionId& id) noexcept;

        /**
         * @brief Updates the status of a client record.
         *
         * Runs `UPDATE clients SET status = ? WHERE id = ?` and returns true
         * if the update was successful.
         *
         * @param id     The client id to update.
         * @param status The new status to set.
         * @return true   if the update was successful.
         * @return false  on prepare/query failure.
         */
        bool updateStatus(const C2Common::Session::sessionId& id,
                          const C2Common::Session::SessionStatus& status) noexcept;

        C2Common::Result<std::vector<C2Common::Session::SessionRecord>> getAll(C2Common::Session::SessionStatus s = C2Common::Session::SessionStatus::CONNECTED)  noexcept;


    private:
        /**
         * @brief Returns the application database connection.
         *
         * Retrieves the `Database` reference from `Context::getInstance()` on
         * every call — no reference is stored on this object.
         */
        C2::db::Database & db() noexcept;

    };

} // namespace CppC2::db
